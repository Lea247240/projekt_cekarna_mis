CREATE TABLE Pacient (
    PacientID INTEGER PRIMARY KEY AUTOINCREMENT,
    Jmeno VARCHAR(100),
    Heslo VARCHAR(50),
    Telefon VARCHAR(20)
);


CREATE TABLE Vykon (
    VykonID INTEGER PRIMARY KEY AUTOINCREMENT,
    PacientID INTEGER NOT NULL,
    TypVykonu VARCHAR(100),
    FOREIGN KEY (PacientID) REFERENCES Pacient(PacientID)
);



CREATE TABLE Cekarna (
    CekarnaID INTEGER PRIMARY KEY AUTOINCREMENT,
    PacientID INTEGER NOT NULL,
    Jmeno VARCHAR(100) NOT NULL,
    Poradi INTEGER NOT NULL,
    FOREIGN KEY (PacientID) REFERENCES Pacient(PacientID)
);


CREATE TABLE CiselnikVysetrovna (
    VysetrovnaID INTEGER PRIMARY KEY AUTOINCREMENT,
    Cislo INTEGER NOT NULL,
    Doktor VARCHAR(100)
);
